\section{Rappels}

\diapo{Rappels}{
\begin{itemize}
\item Réseau de neurones
	\begin{itemize}
	\item Fonction de transfert pour chaque neurone
	\item Relation entre les neurones
	\item Organisation du réseau (couches)
	\end{itemize}
\item Apprentissage non-supervisé
	\begin{itemize}
	\item Le réseau évolue seul en fonction des entrées qui lui sont données.
	\item Il modifie les poids des liaisons entre neurones.
	\end{itemize}
\end{itemize}
}

\diapo{Structure}{
	\input{input/struct.tex}
}

\diapo{Apprentissage}{
	\begin{itemize}
	\item Porte sur les neurones de la carte
	\item On repère le neurone \imp{vainqueur} qui est le plus "proche" de la couche d'entrée au sens des poids.
	\item Ses poids sont modifiés pour diminuer cette distance.
	\item Cette modification est propagée aux neurones voisins du vainqueur.
	\end{itemize}
}

\section{Application}

\diapo{Application : ROC}{
	\begin{itemize}
	\item la couche d'entrée est un agencement en 2 dimensions de neurone représentant chacun un pixel de l'image.
	\item La carte de Kohonen est de dimension 2
	\item L'ensemble des poids d'un neurone de la carte peut être vu comme la moyenne des pixels ayant été vue par ce neurone. Chaque neurone est donc une moyenne d'image qu'il a rencontrées.
	\end{itemize}
	\texttt{http://hurlebouc.github.com/kohonen/doc/inherits.html}
}
